433 class host_exerciser : public test_afu {
434 public:
435  host_exerciser()

437 
438  virtual int run(CLI::App *app, test_command::ptr_t test) override {
439  int res = exit_codes::not_run;
440 
441  logger_->set_pattern(" %v");
442  // Info prints details of an individual run. Turn it on if doing only one
443  // test and the user hasn't changed level from the default.
444  if ((log_level_.compare("warning") == 0))
445  logger_->set_level(spdlog::level::info);
446 
447 
448  logger_->info("starting test run, count of {0:d}", count_);
449  uint32_t count = 0;
450  try {
451  while (count < count_) {
452  logger_->debug("starting iteration: {0:d}", count + 1);
453 
454  res = test_afu::run(app, test);
455  count++;
456  logger_->debug("end iteration: {0:d}", count);
457  if (res)
458  break;
459  }
460  } catch (std::exception &ex) {
461  logger_->error(ex.what());
462  res = exit_codes::exception;
463  }
464 
465  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
466  logger_->info("Test {}({}): {}", test->name(), count, pass);
467  spdlog::drop_all();
468  return res;
469  }
470 
471 public:
472  uint32_t count_;
473 
474  bool option_passed(std::string option_str) {
475  if (app_.count(option_str) == 0)
476  return false;
477  return true;
478  }
479 };
