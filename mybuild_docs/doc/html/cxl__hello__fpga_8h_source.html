377 class hello_fpga : public test_afu {

381  virtual int run(CLI::App *app, test_command::ptr_t test) override {
382  int res = exit_codes::not_run;
383 
384  logger_->set_pattern(" %v");
385  // Info prints details of an individual run. Turn it on if doing only one
386  // test and the user hasn't changed level from the default.
387  if ((log_level_.compare("warning") == 0))
388  logger_->set_level(spdlog::level::info);
389 
390  logger_->info("starting test run, count of {0:d}", count_);
391  uint32_t count = 0;
392  try {
393  while (count < count_) {
394  logger_->debug("starting iteration: {0:d}", count + 1);
395 
396  res = test_afu::run(app, test);
397  count++;
398  logger_->debug("end iteration: {0:d}", count);
399  if (res) break;
400  }
401  } catch (std::exception &ex) {
402  logger_->error(ex.what());
403  res = exit_codes::exception;
404  }
405 
406  auto pass = res == exit_codes::success ? "PASS" : "FAIL";
407  logger_->info("Test {}({}): {}", test->name(), count, pass);
408  spdlog::drop_all();
409  return res;
410  }
411 
412  public:
413  uint32_t count_;
414 
415  bool option_passed(std::string option_str) {
416  if (app_.count(option_str) == 0) return false;
417  return true;
418  }
