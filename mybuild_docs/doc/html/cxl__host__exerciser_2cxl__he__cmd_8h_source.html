418  uint64_t __mem_virt2phys(const void* virtaddr)
419  {
420  int page_size = getpagesize();
421  struct timespec mtime1, mtime2;
422  const char* fname = "/proc/self/pagemap";
423  unsigned long pfn;
424  uint64_t page;
425  off_t offset;
426  int fd, retval;
427 
428  retval = get_mtime(fname, &mtime1);
429  if (retval) {
430  cerr << "stat failed\n";
431  return -1;
432  }
433 
434  fd = file_open("/proc/self/pagemap", O_RDONLY | O_EXCL);
435  if (fd < 0)
436  return -1;
437 
438  pfn = (unsigned long)virtaddr / page_size;
439  offset = pfn * sizeof(uint64_t);
440  if (lseek(fd, offset, SEEK_SET) == (off_t)-1) {
441  cerr << "seek error\n";
442  close(fd);
443  return -1;
444  }
445 
446  retval = get_mtime(fname, &mtime2);
447  if (retval) {
448  cerr << "stat failed\n";
449  close(fd);
450  return -1;
451  }
452 
453  if (mtime1.tv_sec != mtime2.tv_sec
454  || mtime1.tv_nsec != mtime2.tv_nsec) {
455  cerr << "file got modified after open \n";
456  close(fd);
457  return -1;
458  }
459 
460  retval = read(fd, &page, PFN_MASK_SIZE);
461  close(fd);
462  if (retval != PFN_MASK_SIZE)
463  return -1;
464  if ((page & 0x7fffffffffffffULL) == 0)
465  return -1;
466 
467  return ((page & 0x7fffffffffffffULL) * page_size)
468  + ((unsigned long)virtaddr & (page_size - 1));
469  }

471  bool create_linked_list(uint64_t *virt_ptr_a, uint64_t phy_ptr_a,
472  uint64_t data, uint64_t max_size, he_bias_support bias_a = HOSTMEM_BIAS,
473  uint64_t *virt_ptr_b = NULL, uint64_t phy_ptr_b = 0,
474  he_bias_support bias_b = FPGAMEM_HOST_BIAS) {
475 
476  uint64_t *temp_virt_ptr_a = virt_ptr_a;
477  uint64_t temp_phy_ptr_a = phy_ptr_a;
478  uint64_t *temp_virt_ptr_b = virt_ptr_b;
479  uint64_t temp_phy_ptr_b = phy_ptr_b;
480  uint64_t i = 0;
481  struct he_cache_running_ptr *temp_node = NULL;
482 
483  host_exe_->logger_->debug("virt_ptr_a:{:p}", fmt::ptr(virt_ptr_a));
484  host_exe_->logger_->debug("phy_ptr_a:0x{:x}", phy_ptr_a);
485  host_exe_->logger_->debug("virt_ptr_b:{:p}", fmt::ptr(virt_ptr_b));
486  host_exe_->logger_->debug("phy_ptr_b:0x{:x}", phy_ptr_b);
487  host_exe_->logger_->debug("max_size:{0}", max_size);
488  host_exe_->logger_->debug("data:{:x}", data);
489 
490  if (virt_ptr_a == NULL || phy_ptr_a == 0) {
491  cerr << "Invalid input arguments" << endl;
492  return false;
493  }
494 
495  // Linked list on host or fpga memory
496  if (virt_ptr_a != NULL && phy_ptr_a != 0 &&
497  virt_ptr_b == NULL && phy_ptr_b == 0) {
498 
499  temp_node = (struct he_cache_running_ptr*)(temp_virt_ptr_a);
500 
501  for (i = 0; i < max_size; ++i) {
502  temp_node->phy_next_ptr = temp_phy_ptr_a + 64;
503  temp_node->data = data;
504  temp_node->virt_next_ptr = (struct he_cache_running_ptr*)(temp_virt_ptr_a + 8);
505  temp_node->biasmode = bias_a;
506 
507  temp_node++;
508  temp_phy_ptr_a = temp_phy_ptr_a + 64;
509  temp_virt_ptr_a = temp_virt_ptr_a + 8;
510  }
511 
512  temp_node->phy_next_ptr = 0;
513  temp_node->virt_next_ptr = NULL;
514  temp_node->data = 0;
515  }
516 
517  // Linked list on host and fpga memory
518  if (virt_ptr_a != NULL && phy_ptr_a != 0 &&
519  virt_ptr_b != NULL && phy_ptr_b != 0) {
520 
521  struct he_cache_running_ptr* temp_node_a =
522  (struct he_cache_running_ptr*)(temp_virt_ptr_a);
523 
524  struct he_cache_running_ptr* temp_node_b =
525  (struct he_cache_running_ptr*)(temp_virt_ptr_b);
526 
527  int which = 0;
528  for (i = 0; i < max_size; ++i) {
529 
530  if (which == 0) {
531  temp_node_a->phy_next_ptr = temp_phy_ptr_b;
532  temp_node_a->data = data;
533  temp_node_a->biasmode = bias_b;
534  temp_node_a->virt_next_ptr = temp_node_b;
535  ++temp_node_a;
536  temp_phy_ptr_a += 64;
537 
538  } else {
539  temp_node_b->phy_next_ptr = temp_phy_ptr_a;
540  temp_node_b->data = data;
541  temp_node_b->biasmode = bias_a;
542  temp_node_b->virt_next_ptr = temp_node_a;
543  ++temp_node_b;
544  temp_phy_ptr_b += 64;
545  }
546 
547  which = 1 - which;
548  }
549 
550  temp_node_a->phy_next_ptr = 0;
551  temp_node_a->virt_next_ptr = nullptr;
552 
553  temp_node_b->phy_next_ptr = 0;
554  temp_node_b->virt_next_ptr = nullptr;
555  }
556 
557  return true;
558  }
