63  void he_perf_counters(he_cxl_latency cxl_latency = HE_CXL_LATENCY_NONE) {
64  volatile he_cache_dsm_status *dsm_status = NULL;
65 
66  dsm_status = reinterpret_cast<he_cache_dsm_status *>(
67  (uint8_t *)(host_exe_->get_dsm()));
68  if (!dsm_status) return;
69 
70  cout << "\n********* DSM Status CSR Start *********" << endl;
71  cout << "test completed :" << dsm_status->test_completed << endl;
72  cout << "dsm number:" << dsm_status->dsm_number << endl;
73  cout << "error vector:" << dsm_status->err_vector << endl;
74  cout << "num ticks:" << dsm_status->num_ticks << endl;
75  cout << "num reads:" << dsm_status->num_reads << endl;

77  cout << "penalty start:" << dsm_status->penalty_start << endl;
78  cout << "penalty end:" << dsm_status->penalty_end << endl;
79  cout << "actual data:" << dsm_status->actual_data << endl;
80  cout << "expected data:" << dsm_status->expected_data << endl;
81 
82  // print bandwidth
83  if (dsm_status->num_ticks > 0) {
84  double perf_data =
85  he_num_xfers_to_bw(dsm_status->num_reads + dsm_status->num_writes,
86  dsm_status->num_ticks);
87  host_exe_->logger_->info("Bandwidth: {0:0.3f} GB/s", perf_data);
88  }
89 
90  if (cxl_latency == HE_CXL_RD_LATENCY) {
91  if (dsm_status->num_ticks > 0 && dsm_status->num_reads > 0) {
92  double latency =
93  (double)((dsm_status->num_ticks / (double)dsm_status->num_reads) *
94  (2.5));
95 
96  host_exe_->logger_->info("Read Latency : {0:0.2f} nanoseconds",
97  latency);
98  } else {
99  host_exe_->logger_->info("Read Latency: N/A");
100  }
101  }
102 
103  cout << "********* DSM Status CSR end *********" << endl;
104  }

244  bool he_set_bias_mode() {
245  // Target memory HOST set BIAS host
246  if (he_target_ == HE_TARGET_HOST) {
247  he_ctl_.bias_support = HOSTMEM_BIAS;
248  // Target memory FPGA set BIAS host/device
249  if (he_bias_ == HOSTMEM_BIAS) {
250  he_ctl_.bias_support = HOSTMEM_BIAS;
251  } else {
252  cerr << "Wrong BIAS mode for specified target memory type" << endl;
253  return false;
254  }
255  } else {
256  // Target memory FPGA set BIAS host/device
257  if (he_bias_ == HOSTMEM_BIAS) {
258  he_ctl_.bias_support = FPGAMEM_HOST_BIAS;
259  } else {
260  he_ctl_.bias_support = FPGAMEM_DEVICE_BIAS;
261  }
262  }
263 
264  return true;
265  }
