70  bool hello_fpga_data_intg_check(uint8_t *buf_address) {
71  uint8_t *data_buff_addr_ = NULL;
72  uint16_t hw_CL = 0;
73  uint32_t cl_idx = 0;
74  uint8_t idx = 0;
75  uint32_t header_data_ = 0;
76  uint64_t hw_phy_addr_ = 0;
77  volatile uint64_t *act_wr_buff_phy_addr = NULL;
78 
79  act_wr_buff_phy_addr = host_exe_->get_write_buff_phy_addr();
80 
81  for (cl_idx = 0; cl_idx < (HELLO_FPGA_NUMCACHE_LINES); cl_idx++) {
82  data_buff_addr_ = (buf_address + (CL * cl_idx));
83 
84  hw_CL = (((uint16_t)data_buff_addr_[0]) |
85  ((uint16_t)(data_buff_addr_[1]) << 8));
86 
87  if (hw_CL != cl_idx) {
88  cout << "\n Cache Line counter mismatch " << endl;
89  host_exe_->free_cache_read_write();
90  host_exe_->free_dsm();
91  return false;
92  }
93 
94  header_data_ = (data_buff_addr_[2] | (data_buff_addr_[3] << 8) |
95  (data_buff_addr_[4] << 16) | ((data_buff_addr_[5] & 0x3F) << 24));
96 
97  if (HELLO_FPGA_CL_HEADER != header_data_) {
98  cout << "\n Cache Line Header value mismatch - " << header_data_
99  << endl;
100  host_exe_->free_cache_read_write();
101  host_exe_->free_dsm();
102  return false;
103  }
104 
105  if (0 != ((data_buff_addr_[5] >> 6) & 0x3)) {
106  cout << "\n Stride value mismatch " << endl;
107  host_exe_->free_cache_read_write();
108  host_exe_->free_dsm();
109  return false;
110  }
111 
112  for (idx = 6; idx < 56; idx++) {
113  if (0 != data_buff_addr_[idx]) {
114  cout << "\n A non-ZERO value is observed at byte " << idx << endl;
115  host_exe_->free_cache_read_write();
116  host_exe_->free_dsm();
117  return false;
118  }
119  }
120 
121  hw_phy_addr_ = ((uint64_t)(data_buff_addr_[56]) |
122  ((uint64_t)(data_buff_addr_[57]) << 8) |
123  ((uint64_t)(data_buff_addr_[58]) << 16) |
124  ((uint64_t)(data_buff_addr_[59]) << 24) |
125  ((uint64_t)(data_buff_addr_[60]) << 32) |
126  ((uint64_t)(data_buff_addr_[61]) << 40) |
127  ((uint64_t)(data_buff_addr_[62]) << 48) |
128  ((uint64_t)(data_buff_addr_[63]) << 56));
129 
130  if (*act_wr_buff_phy_addr != hw_phy_addr_) {
131  cout << "\n Cache Line Physical Address mismatch " << endl;
132  host_exe_->free_cache_read_write();
133  host_exe_->free_dsm();
134  return false;
135  }
136 
137  *act_wr_buff_phy_addr = *act_wr_buff_phy_addr + CL;
138  }
139  return true;
140  }

142  int he_run_hello_fpga_test() {
143  cout << "Hello FPGA Start" << endl;
144 
145  uint8_t *data_buff_addr = NULL;
146 
147  /*
148  STEPS
149  1) Allocate DSM, Read buffer, Write buffer
150  2) Write number of lines more then 32 kb 2mb/64
151  3) Set WR ItoMWr (CXL) config
152  4) Run test ( Buffer is not present in FPGA - FPGA write Cache miss )
153  */
154 
155  /*
156  HE_INFO
157  Set Read number Lines
158  */
159  he_info_.value = host_exe_->read64(HE_INFO);
160  host_exe_->write64(HE_WR_NUM_LINES, HELLO_FPGA_NUMCACHE_LINES);
161 
162  cout << "Read/write number Lines:" << HELLO_FPGA_NUMCACHE_LINES << endl;
163  cout << "Read address table size:" << he_info_.read_addr_table_size << endl;
164  cout << "Write address table size:" << he_info_.write_addr_table_size
165  << endl;
166 
167  /*
168  * set W_CONFIG
169  */
170  he_wr_cfg_.value = 0;
171  he_wr_cfg_.write_traffic_enable = 1;
172  he_wr_cfg_.opcode = WR_LINE_M;
173  host_exe_->write64(HE_WR_CONFIG, he_wr_cfg_.value);
174 
175  /*
176  * Set WR_ADDR_TABLE_CTRL
177  */
178  wr_table_ctl_.value = 0;
179  wr_table_ctl_.enable_address_stride = 1;
180  host_exe_->write64(HE_WR_ADDR_TABLE_CTRL, wr_table_ctl_.value);
181 
182  /*
183  * Allocate DSM buffer
184  */
185  if (!host_exe_->allocate_dsm()) {
186  cerr << "allocate dsm failed" << endl;
187  return -1;
188  }
189 
190  /*
191  * Allocate Read, Write buffer
192  */
193  if (!host_exe_->allocate_cache_read_write(BUFFER_SIZE_2MB, numa_node_)) {
194  cerr << "allocate cache read failed" << endl;
195  host_exe_->free_dsm();
196  return -1;
197  }
198 
199  /*
200  * Obtain Write address
201  */
202  data_buff_addr = host_exe_->get_read_write();
203 
204  memset(data_buff_addr, 0xFF, BUFFER_SIZE_2MB);
205 
206  /*
207  * Start test
208  */
209  he_start_test();
210 
211  cout << "AFU Configuration : Successful" << endl;
212  /*
213  * wait for completion
214  */
215  if (!he_wait_test_completion()) {
216  he_perf_counters();
217  host_exerciser_errors();
218  host_exe_->free_cache_read_write();
219  host_exe_->free_dsm();
220  return -1;
221  }
222 
223  if (!hello_fpga_data_intg_check(data_buff_addr)) {
224  host_exe_->free_cache_read_write();
225  host_exe_->free_dsm();
226  cout << "DATA Integrity Check : Failed" << endl;
227  return -1;
228  }
229  cout << "DATA Integrity Check : Successful" << endl;
230 
231  host_exe_->free_cache_read_write();
232  host_exe_->free_dsm();
233 
234  cout << "Hello FPGA Executed Successfully" << endl;
235  return 0;
236  }
